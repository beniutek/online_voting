def create
  valid, i = vote_service.count_vote(signature: signature_param, message: message_param)
  if valid
    render json: {
      index: i,
      bit_commitment: message_param,
      signature: signature_param
    }
  else
    render json: {
      error: i,
    }, status: 400
  end
end
